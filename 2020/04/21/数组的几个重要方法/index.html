forEach

forEa使函数作用在每个数组元素上，但是没有返回值。

例子：//给每个人添加一个函数，返回一个班级属性

    let arr = [
        {name:'xiaoming',grade:99},
        {name:'xiaohei',grade:89},
        {name:'xiaolong',grade:59}
    ];
    arr.forEach(function(item,index,array){
        return item['class'] = '一年级一班';
    })

reduce

使用 reduce 函数可以从前面开始迭代数组的所有元素

参数 说明
pre 调用上次返回的结果（首次则为0）
cur 当前元素值
index 当前索引
array 原数组 
function countNumber(arry,item){
    return arry.reduce(function(pre,cur){
        return pre+= cur == item ? 1:0
    },0)
}
let arr = [1,2,3,4,5,62,1,1,1]
console.log(countNumber(arr,1));//统计arr里的1的数量

例子：//找出最大值

function arrMax(array){
    return array.reduce(function(max,cur){
        return max > cur ? max:cur 
    },[])
}
let arr = [1,2,3,4,5,62,1,1,1]
console.log(arrMax(arr));
